@startuml

enum ChannelCountry {
  RU
  US
  UK
  ...
}

enum LikeType {
  LIKE
  DISLIKE
}

enum LikeTarget {
  VIDEO
  COMMENT
}

class User {
  + userId: UUID
  + username: String
  + password: String
  
  + register()
  + login()
}

class Channel {
  + channelId: UUID
  + name: String
  + description: String
  + country: ChannelCountry
  
  + createChannel()
  + deleteChannel()
  + getInfo()
}

class Subscription {
  + subscriptionId: UUID
  + userId: UUID
  + channelId: UUID
  + subscriptionTime: DateTime
  
  + subscribe()
  + unsubscribe()
}

class Video {
  + videoId: UUID
  + slug: String
  + title: String
  + description: String
  + createdAt: DateTime
  
  + uploadVideo()
  + deleteVideo()
  + viewVideo()
  + searchVideos()
}

class View {
  + viewId: UUID
  + videoId: UUID
  + userId: UUID
}

class Playlist {
  + playlistId: UUID
  + name: String
  + userId: UUID
  
  + createPlaylist()
  + viewPlaylist()
}

class Comment {
  + commentId: UUID
  + videoId: UUID
  + userId: UUID
  + body: String
  
  + sendComment()
  + updateComment()
}

class Like {
  + likeId: UUID
  + userId: UUID
  + targetType: LikeTarget
  + targetId: UUID
  + type: LikeType
  
  + like()
  + dislike()
  + removeLike()
}

@enduml
